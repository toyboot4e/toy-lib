msbOf :: Int -> Int #

Retrieves the most significant bit.

>>> msbOf 0
-1
>>> msbOf maxBound
62
>>> msbOf $ 4 + 2 + 1
2

lsbOf :: Int -> Int #

Retrieves the least significant bit.

>>> lsbOf 0
64
>>> lsbOf maxBound
0
>>> lsbOf $ 4 + 2 + 1
0
