msbOf :: Int -> Int #

Retrieves the most significant bit.

>>> msbOf 0
-1
>>> msbOf maxBound
62
>>> msbOf $ 4 + 2 + 1
2

lsbOf :: Int -> Int #

Retrieves the least significant bit.

>>> lsbOf 0
64
>>> lsbOf maxBound
0
>>> lsbOf $ 4 + 2 + 1
0

log2 :: FiniteBits b => b -> Int #

Log base of two or bit floor. https://hackage.haskell.org/package/base-4.17.0.0/docs/Data-Bits.html#v:countLeadingZeros

log2CeilInt :: Int -> Int #

Ceiling of log base 2 of an Int.

Example

> log2 3
1
> log2CeilInt 3
2

bitCeil :: Int -> Int #

Calculates the smallest integral power of two that is not smaller than x.

Example

> bitCeil 3
4
