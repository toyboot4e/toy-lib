{-# LANGUAGE BangPatterns #-}
{-# LANGUAGE CPP #-}

module ToyLib.Macro where

-- When run as a stack script, `dbg` expands to `traceShow`.
-- Otherwise it's an empty function.
#ifdef DEBUG
dbg :: Show a => a -> ()
dbg !x = let !_ = traceShow x () in ()

dbgAssert :: Bool -> String -> ()
dbgAssert False !s = error $ "assertion failed!: " ++ s
dbgAssert True _ = ()

#else
dbg :: Show a => a -> ()
dbg :: forall a. Show a => a -> ()
dbg a
_ = ()

dbgAssert :: Bool -> a -> a
dbgAssert :: forall a. Bool -> a -> a
dbgAssert = forall a b c. (a -> b -> c) -> b -> a -> c
flip forall a b. a -> b -> a
const

#endif
